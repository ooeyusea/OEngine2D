INCLUDE_DIRECTORIES(
	"${PROJECT_SOURCE_DIR}/thirdparty/include"
    "${CMAKE_CURRENT_SOURCE_DIR}/util/"
)

SET(KERNEL
    "${CMAKE_CURRENT_SOURCE_DIR}/kernel.h"
    "${CMAKE_CURRENT_SOURCE_DIR}/kernel.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/main.cpp"
)

SET(PUBLIC
    "${CMAKE_CURRENT_SOURCE_DIR}/util/util.h"
    "${CMAKE_CURRENT_SOURCE_DIR}/util/util.cpp"
)

IF(WIN32)
SET(PUBLIC
	${PUBLIC}
	"${CMAKE_CURRENT_SOURCE_DIR}/util/win32.h"
    "${CMAKE_CURRENT_SOURCE_DIR}/util/win32.cpp"
)
ENDIF(WIN32)

IF(UNIX)
SET(PUBLIC
	${PUBLIC}
	"${CMAKE_CURRENT_SOURCE_DIR}/util/linux.h"
    "${CMAKE_CURRENT_SOURCE_DIR}/util/linux.cpp"
)
ENDIF(UNIX)

SOURCE_GROUP(util FILES ${UTIL})
SOURCE_GROUP(src FILES ${KERNEL})

ADD_EXECUTABLE(OEngine2D
    ${UTIL}
    ${KERNEL}
)

IF(WIN32)
TARGET_LINK_LIBRARIES(OEngine2D 
	debug "${PROJECT_SOURCE_DIR}/thirdparty/libs/debug/glfw3dll.lib"
	optimized "${PROJECT_SOURCE_DIR}/thirdparty/libs/release/glfw3dll.lib"
)
ENDIF(WIN32)

SET(EXECUTABLE_OUTPUT_PATH "${PROJECT_SOURCE_DIR}/bin")
